dzis=new Date() --> bierze z przeglądarki pełną datę i strefe czasową i przypisuje do wartosci dzis
dzis = dzis.getHour --> z pełnej daty wycina godzinę i przypisuje do wartosci dzis
dzis = dzis.getMinutes --> z pełnej daty wycina minuty i przypisuje do wartosci dzis
dzis = dzis.getMonth+1 --> z pełnej daty wycina miesiąc, dodaje 1 ponieważ funkcja liczy daty od 0 i przypisuje do wartosci dzis
dzis = dzis.getDate() --> z pełnej daty wycina numer dnia i przypisuje do wartosci dzis
dzis = dzis.getFullYear --> z pełnej daty wycina rok w formacie rrrr i przypisuje do wartosci dzis
